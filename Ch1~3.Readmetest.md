# 2차원 배열 예시
- #### 사용 상황: 
학생들의 시험 점수를 저장하고 출력하는 경우 (행은 학생, 열은 과목).

$ public class TwoDimensionalArray {

    public static void main(String[] args) {

        int[][] scores = {
            {85, 90, 92}, // 학생 1의 점
            (국어, 수학, 영어)
            {78, 88, 91}, // 학생 2의 점수
            {90, 77, 89}  // 학생 3의 점수
        };

        System.out.println("학생 1의 수학 점수: " + scores[0][1]); // 출력: 90
        System.out.println("학생 2의 영어 점수: " + scores[1][2]); // 출력: 91
    }
}
